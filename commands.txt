
# Composer
 `# Composer - Dump autoload #` composer dump-autoload -o
 `# Composer - Install #` composer install --no-dev
 `# Composer - Update #` composer update


# Docker-compose
 `# docker-compose - Recreate single service #` echo "Service name: "; read service && docker-compose stop $service; docker-compose rm -f $service; docker-compose up -d $service
 `# docker-compose - Log service #` echo "Service name: "; read service && while true; do docker-compose logs --tail 10 -f $service; sleep 2; done
 `# docker-compose - Force service up #` echo "Service name: "; read service && docker-compose up -d --force-recreate $service
