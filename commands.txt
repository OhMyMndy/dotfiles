# Docker-compose
 `# docker-compose - Recreate single service #` if [ "$service" = '' ]; then echo "Service name: "; read service; fi; docker-compose stop $service; docker-compose rm -f $service; docker-compose up -d $service
 `# docker-compose - Log service #` if [ "$service" = '' ]; then echo "Service name: "; read service; fi; while true; do docker-compose logs --tail $COLUMNS -f $service; sleep 2; done
 `# docker-compose - Force service up #` if [ "$service" = '' ]; then echo "Service name: "; read service; fi; docker-compose up -d --force-recreate $service
